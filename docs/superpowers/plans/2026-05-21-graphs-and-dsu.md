void bfs(const vector<vector<int>>& adj, int start) {

    vector<bool> visited(n, false);

void dfsRecursive(const vector<vector<int>>& adj, int u, vector<bool>& visited) {
    visited[u] = true;
    cout << "Visit " << u << "\n";
    for (int v : adj[u]) {
        if (!visited[v]) dfsRecursive(adj, v, visited);
    }
}

void dfsIterative(const vector<vector<int>>& adj, int start) {
    int n = adj.size();
    vector<bool> visited(n, false);
    stack<int> s;
    s.push(start);
    while (!s.empty()) {
        int u = s.top();
        s.pop();
        if (visited[u]) continue;
        visited[u] = true;
        cout << "Visit " << u << "\n";
        // push in reverse so smallest neighbor is popped first
        for (auto it = adj[u].rbegin(); it != adj[u].rend(); ++it) {
            if (!visited[*it]) s.push(*it);
        }
    }
}

int main() {
    vector<vector<int>> adj(5);
    adj[0] = {1, 4};
    adj[1] = {0, 2, 3, 4};
    adj[2] = {1, 3};
    adj[3] = {1, 2, 4};
    adj[4] = {0, 1, 3};

    int n = adj.size();

```cpp
#include <iostream>
#include <vector>
using namespace std;

struct DSU {
    vector<int> p, r;
    DSU(int n) : p(n), r(n, 0) {
        for (int i = 0; i < n; i++) p[i] = i;
    }
    int find(int x) {
        return p[x] == x ? x : p[x] = find(p[x]);  // path compression
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;  // union by rank
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

int main() {
    DSU d(8);
    d.unite(0, 1);
    d.unite(2, 3);
    d.unite(4, 5);
    d.unite(6, 7);
    d.unite(0, 2);  // merges {0,1} ∪ {2,3}
    d.unite(4, 6);  // merges {4,5} ∪ {6,7}
    d.unite(0, 4);  // merges all
    cout << "find(7) = " << d.find(7) << "\n";        // 0 (or whichever root)
    cout << "rank[0] = " << d.r[d.find(0)] << "\n";
    return 0;
}
```

- [ ] **Step 2: Author `graph-bfs` deck**

Key content:
- Title: `Breadth-First Search` / `廣度優先搜尋`
- Core concept: queue-based, explores by distance from source.
- Operation flow: enqueue source → loop: dequeue, visit, enqueue unvisited neighbors. Mermaid: layer-by-layer expansion from source 0.
- SVG: 5-node graph annotated with discovery distances (0, 1, 1, 2, 2).
- Complexity: $O(V + E)$ time, $O(V)$ space.
- Code: `graph_bfs.cpp` `bfs` function.
- Pros/cons: shortest unweighted path, but uses more memory than DFS for wide graphs.
- Summary: queue + visited set; finds shortest paths in unweighted graphs.

- [ ] **Step 3: Author `graph-dfs` deck**

Key content:
- Title: `Depth-First Search` / `深度優先搜尋`
- Core concept: stack-based or recursive, goes deep before backtracking.
- Operation flow: push source → loop: pop, visit, push unvisited neighbors (reverse order). Or recursive: visit, recurse on each unvisited neighbor.
- SVG: 5-node graph with DFS tree edges highlighted vs back/forward/cross edges.
- Complexity: $O(V + E)$ time, $O(h)$ space (recursion/stack depth).
- Code: `graph_dfs.cpp` recursive + iterative variants.
- Pros/cons: small memory for deep graphs; foundation for topological sort, SCC, cycle detection.
- Summary: stack/recursion + visited set; explores paths fully before backtracking.

- [ ] **Step 4: Author `tree-dsu` deck**

Key content:
- Title: `Disjoint Set (Union-Find)` / `不相交集合(並查集)`
- Core concept: forest representation of disjoint sets; `find` follows parent pointers to root; `union` merges two sets by attaching one root to the other.
- Operation flow: with path compression + union by rank, both `find` and `union` run in near-constant amortized time.
- Mermaid: a small forest before/after a couple of unions.
- SVG: tree-of-arrays diagram showing parent pointers + rank array.
- Complexity: amortized $O(\alpha(N))$ per operation (inverse Ackermann, effectively constant); space $O(N)$.
- Code: `tree_dsu.cpp` `struct DSU`.
- Pros/cons: extremely fast in practice; used in Kruskal's MST, network connectivity, image segmentation.
- Summary: parent[] array; path compression + union by rank → near-O(1) amortized; the workhorse for Kruskal.
